String Functions

1. strlen()

The strlen() function is used to find the length of a string. It counts the number of characters in the string but does not include the null character '\0'. 


#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str[] = "Hello World";
    cout << "Length of string = " << strlen(str);
    return 0;
}

2. strcpy()

The strcpy() function copies one string into another. The destination string must have enough space to hold the copied content. 


#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str1[] = "C++ Strings";
    char str2[20];
    strcpy(str2, str1);
    cout << "Copied string: " << str2;
    return 0;
}

3. strcat()

The strcat() function is used to concatenate (join) two strings. The result is stored in the first string, which should have enough memory to hold the final combined string. 


#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str1[30] = "Hello";
    char str2[] = " World";
    strcat(str1, str2);
    cout << "Concatenated string: " << str1;
    return 0;
}

4. strcmp()

The strcmp() function is used to compare two strings. It returns:
  • 0 if both strings are equal
  • Less than 0 if the first string is smaller
  • Greater than 0 if the first string is larger


#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str1[] = "Apple";
    char str2[] = "Banana";
    int result = strcmp(str1, str2);

    if(result == 0)
        cout << "Strings are equal";
    else if(result < 0)
        cout << "str1 is smaller than str2";
    else
        cout << "str1 is greater than str2";

    return 0;
}

5. strchr()

The strchr() function finds the first occurrence of a character in a string. It returns a pointer to the location of the character if found, otherwise NULL. 


#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str[] = "Hello World";
    char *ptr = strchr(str, 'W');
    if(ptr)
        cout << "Character found at position: " << (ptr - str);
    else
        cout << "Character not found";
    return 0;
}

6. strstr()

The strstr() function is used to find the first occurrence of a substring in a string. If the substring is found, it returns a pointer to its first occurrence. 


#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str[] = "I love programming in C++";
    char *ptr = strstr(str, "programming");
    if(ptr)
        cout << "Substring found: " << ptr;
    else
        cout << "Substring not found";
    return 0;
}
